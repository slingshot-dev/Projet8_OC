package tourGuide.Modeles;
import Modeles.Location;

import java.util.Date;
import java.util.UUID;


public class VisitedLocationMain {

    public final UUID userId;
    public final Location locationMain;
    public final Date timeVisited;


    public VisitedLocationMain(UUID userId, Location locationMain, Date timeVisited) {
        this.userId = userId;
        this.locationMain = locationMain;
        this.timeVisited = timeVisited;
    }

    public UUID getUserId() {
        return userId;
    }

    public Location getLocationMain() {
        return locationMain;
    }

    public Date getTimeVisited() {
        return timeVisited;
    }
}
